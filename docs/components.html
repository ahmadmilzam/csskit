Component Intro

CSSKit is a design-free, OOCSS framework. It does its best to provide zero cosmetic styling. This means that CSSKit can be used on any and all types of project without dictating (or even suggesting) a look-and-feel. If you do require a UI out of the box, then CSSKit is probably not the best tool for you. Iâ€™d recommend looking at a UI Toolkit such as Foundation or Bootstrap.

Because CSSKit does no cosmetic styling, it is up to you to author the Components layer. Components are small partials that contain discrete chunks of UI that utilise the layers that came before it, for example, a carousel, a tabbed nav, pagination or an image gallery, and so on.

However, I will show you how to make your own UI Components build on top CSSKit's Module. It's Easy!

/* ==========================================================================
   Pagination Build on top CSSKit
   ========================================================================== */

// Here we set a variable assuming that default and module settings
// are not set globally. If they have been previously been defined,
// the following variable will be overriden and will be set to `true`.
$imported-default-settings: false !default;

@if ($imported-default-settings == false) {
  @warn "Oops! Have you included default settings and module settings files?"
}


// VARS
$pagination-namespace:          $namespace !default;
$pagination-border-color:       darken($base-background-color, 10%) !default;
$pagination-border-thickness:   1px !default;
$pagination-padding:            rem-calc(quarter($base-spacing-unit)) !default;
$pagination-padding--large:     rem-calc(halve($base-spacing-unit)) !default;
$pagination-link-color:         $base-text-color !default;
$pagination-active-link-color:  $color-white !default;
$pagination-active-bg:          $color-primary;

.#{$pagination-namespace}pagination{
  @extend %#{$list-inline-namespace}list-inline;
}
  .#{$pagination-namespace}pagination__item{
    @extend  %#{$list-inline-namespace}list-inline__item;
  }

  .#{$pagination-namespace}pagination__link{
    display: inline-block;
    color: $pagination-link-color;
    border: $pagination-border-thickness solid $pagination-border-color;
    border-radius: $base-radius;
    padding: $pagination-padding $pagination-padding*2;

    &:hover,
    &:active{
      color: $pagination-link-color;
      background-color: lighten($pagination-active-bg, 50%);
    }

    &.is-disabled{
      color: rgba($pagination-link-color, 0.4);
      pointer-events: none;
    }

    &.is-current{
      border-color: $pagination-active-bg;
      color: $pagination-active-link-color;
      background-color: $pagination-active-bg;
    }

    &.is-empty{
      border-width: 0;
    }
// Only show pagination button (prev, next, first, last) in small viewport
@media screen and (max-width: 720px){
  .#{$pagination-namespace}pagination--responsive .#{$pagination-namespace}pagination__link:not(.is-button){
    display: none;
  }
}

/* ==========================================================================
   TABS build on top CSSKit
   ========================================================================== */

// Here we set a variable assuming that default and module settings
// are not set globally. If they have been previously been defined,
// the following variable will be overriden and will be set to `true`.
$imported-default-settings: false !default;
$imported-module-settings: false !default;

@if ($imported-default-settings == false) {
  @warn "Oops! Have you included default settings and module settings files?"
}

// VARS
$tabs-namespace:        $namespace !default;
$tabs-border-color:     #ddd;
$tabs-border-thickness: 1px;
$tabs-link-padding:     $base-spacing-unit !default;
$tabs-link-color:       $base-text-color !default;
$tabs-link-bg:          darken($base-background-color, 2.5%) !default;
$tabs-link-hover-bg:    darken($base-background-color, 5%) !default;
$tabs-link-active-bg:   $base-background-color !default;

.#{$tabs-namespace}tabs-container{
  border-bottom: $tabs-border-thickness solid $tabs-border-color;
}

/**
 * 1. Reset any residual styles by extend module list inline.
 */
.#{$tabs-namespace}tabs{
  @extend %#{$list-inline-namespace}list-inline; /* [1] */
  // padding-left: rem-calc(halve($base-spacing-unit));
}

  .#{$tabs-namespace}tabs__item{
    @extend %#{$list-inline-namespace}list-inline__item; /* [1] */
  }

    .#{$tabs-namespace}tabs__link{
      display: block;
      padding: rem-calc(halve($tabs-link-padding) $tabs-link-padding);
      border-top-width: $tabs-border-thickness;
      border-bottom-width: 0;
      border-left-width: $tabs-border-thickness;
      border-right-width: $tabs-border-thickness;
      border-style: solid;
      border-color: $tabs-border-color;
      border-radius: $base-radius $base-radius 0 0;
      background-color: $tabs-link-bg;
      color: $tabs-link-color;
      text-align: center;

      &:hover,
      &:active{
        color: $tabs-link-color;
        background-color: $tabs-link-hover-bg;
      }

      &.is-active{
        border-color: darken($tabs-border-color, 10%);
        border-bottom-width: $tabs-border-thickness;
        border-bottom-color: $tabs-link-active-bg;
        background-color: $tabs-link-active-bg;
        margin-bottom: -$tabs-border-thickness;
      }
    }
